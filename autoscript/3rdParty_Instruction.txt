## HDF5 Build
tar xvf hdf5-1.8.22.tar.gz
mkdir hdf5-1.8.22-install
cd hdf5-1.8.22

## CGNS BUILD

mkdir CGNS-4.2.0-install

# Only build basic
cmake .. -DCGNS_ENABLE_HDF5=ON -DCGNS_ENABLE_LEGACY=ON -DCMAKE_INSTALL_PREFIX=../../CGNS-4.2.0-install

# build cgns tools and tests
